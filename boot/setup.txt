# Initial Konfiguration
# Pi-Ager 

# Standard Hostname              Pi-Ager 
# Standard Passwoerter für alle  raspberry
# 
# Nur ausgefüllte Werte werden gesetzt
# Die Variablennamen dürfen nicht verändert werden!
# Wenn die Werte Sonderzeichen wie /?}][{|@!"§$%&-_.:,;#+~* oder Leerzeichen enthalten,
# müssen die Werte in 'Single Quotes' gesetzt werden. 'Single Quotes' dürfen dann nicht
# Bestandteil des Wertes sein!
#
# Z.B.: wlankey='FRITZ!Box 7490 |&;()<>"\$`^~@{}.'
#
# Logfile wird unter /boot/setup.log angelegt 

# Hostname, wie soll der Raspberry Pi im Netzwerk heissen:
piname='pi-ager'

# Passwort für Benutzer pi:
pipass='raspberry'

# Passwort für Benutzer root:
rootpass='raspberry'

# Passwort für Benutzer pi-ager auf der Webiste settings, admin, webcam:
webguipw='raspberry'

# Passwort für Zugriff auf Datenbankverwaltung phpliteadmin:
dbpw='raspberry'

#--------------------------------------------------------------------------------
# Pi-Ager funktioniert auch ohne Router und ohne Internet Verbindung.
# Hat der Pi-Ager keine Verbindung zu einem WLAN Router oder Access Point, 
# schaltet der Pi-Ager automatisch in den Hotspot Mode.
# Er kann dann via Smartphone oder Tablet erreicht werden.
# Der Pi-Ager Hotspot hat die WLAN SSDI 'RPiHotspot',
# das Passwort ist '1234567890'.
# Der Pi-Ager ist dann über die IP Adresse 10.0.0.5 via SSH ereichbar,
# den Pi-Ager Webserver kann man mit einem Browser unter der URL http://10.0.0.5/ erreichen.
#
# Die Überprüfung, ob eine Verbindung zu einem Router oder Access Point möglich ist,
# erfolgt automatisch alle 2 Minuten.
#
# Im Hotspot Mode können auf der ADMIN Seite Datum und Uhrzeit gesetzt werden.
# Sollte später ein WLAN Router verfügbar sein, ist es auch möglich,
# WLAN SSID und WLAN Passwort des Routers einzugeben.
#-------------------------------------------------------------------------------

# WLAN Netzwerkname:
wlanssid=

# WLAN Schlüssel (mindesten 8 Zeichen):
wlankey=

# ISO code your country, e.g. DE, GB:
country='DE'

# Temperature/Humidity sensor attched to pi-ager:
# Supported types are : DHT11, DHT22, SHT75, SHT85, SHT3x
sensor='SHT3x'

# hmi display device name:
# valid device names: NX3224K028 or NX3224T028 or NX3224F028
# z.B. : hmidisplay='NX3224K028'
hmidisplay=

# setup.txt nach boot löschen? leer lassen für löschen oder 1 für behalten 
keepconf=1
